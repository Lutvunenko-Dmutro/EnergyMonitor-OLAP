    lm.timestamp,
    r.region_name,
    lm.actual_load_mw,
    wr.temperature
FROM LoadMeasurements lm

WHERE lm.timestamp >= NOW() - INTERVAL '30 days'
ORDER BY lm.timestamp ASC;
